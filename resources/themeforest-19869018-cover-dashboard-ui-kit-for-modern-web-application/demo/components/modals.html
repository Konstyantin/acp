Form Modal

Using another component as modal's content. 

<modal v-model="formModal">
    <form action="#" method="POST">
        <div class="c-modalCard">
            <header class="c-modalCard__header">
                <h4 class="c-modalCard__title">Create Channel</h4>
            </header>
            <div class="c-modalCard__body">
                <div class="form-group">
                    <label class="form-label">Name</label>
                    <input type="text" class="form-input" name="name" autofocus>
                </div>
            </div>
            <footer class="c-modalCard__footer">
                <button type="button" class="btn btn--default" v-on:click="formModal = false">Cancel</button>
                <button type="submit" class="btn btn--primary">Create</button>
            </footer>
        </div>
    </form>
</modal>

Sizes

By adding size="lg" or size="sm" attribute on <modal> container to change modal content sizes 

<modal v-model="smShow" size="sm">
    <div class="c-modalCard">
        // ...
    </div>
</modal>

<modal v-model="largeShow" size="lg">
    <div class="c-modalCard">
        // ...
    </div>
</modal>

<button class="btn btn--primary" v-on:click="largeShow = true">Large</button>
<button class="btn btn--primary" v-on:click="smShow = true">Small</button>

<script>
    new Vue({
        el: '#app',
        data: {
            smShow: false,
            largeShow: false
        }
    })
</script>
